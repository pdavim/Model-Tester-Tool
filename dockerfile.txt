# Stage 1: Build the frontend and backend
FROM node:20-alpine AS builder

WORKDIR /app

# Install dependencies needed for build
COPY package*.json ./
RUN npm install

# Copy all source files
COPY . .

# Build the frontend (creates /app/dist)
RUN npm run build

# Stage 2: Production Runner
FROM node:20-alpine AS runner

WORKDIR /app

# Set node environment
ENV NODE_ENV=production

# Install only production dependencies
COPY package*.json ./
RUN npm install --omit=dev

# Copy the built frontend from stage 1
COPY --from=builder --chown=node:node /app/dist ./dist

# Copy server-side source code and configuration
COPY --chown=node:node server.ts ./
COPY --chown=node:node tsconfig.json ./
COPY --chown=node:node src ./src
COPY --chown=node:node env.example.txt ./.env.template

# Ensure the app directory belongs to the node user
RUN chown -R node:node /app

# Expose the application port
EXPOSE 3767

# Use a non-root user for security
USER node

# Start the server using tsx as defined in package.json start script
CMD ["npm", "start"]
